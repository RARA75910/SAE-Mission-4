SELECT Fournisseur.Contact, COUNT() AS NombreDeBouteilles FROM Fournisseur INNER JOIN Produit ON Fournisseur.NoFour = Produit.NoFour WHERE Produit.QteParUnite LIKE "bouteilles*" GROUP BY Fournisseur.Contact ORDER BY Fournisseur.Contact;

Indicateur 2 : clients qui n'ont pas encore commmandé




listes des données des clients qui n'ont pas finalisé leur commande.

SELECT Client.*
FROM Client
LEFT JOIN Commande ON Client.CodeCli = Commande.CodeCli
WHERE Commande.NoCom IS NULL;

Indicateur 3: produit conditionnés dans du verre ou carton




Les produits conditionnés dans du carton ou dans du verre on appliqué cette commande :




SELECT Produit.RefProd, Produit.NomProd, Produit.QteParUnite
FROM Produit
WHERE (((Produit.QteParUnite) Like "*carton*" Or (Produit.QteParUnite) Like "*verre*"));


Indicateur 4 : les trois villes qui ont la clientèle la plus importantes




SELECT TOP 3 VilleCli, COUNT(*) AS NombreClients
FROM Client
GROUP BY VilleCli
ORDER BY COUNT(*) DESC;

le resultat est : Agen, Paris, Nantes et Lille avec 3 clients
